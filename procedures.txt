CREATE PROCEDURE sp_SearchCustomers
    @SearchTerm NVARCHAR(100)
AS
BEGIN
    SET NOCOUNT ON;
    
    SELECT TOP 15
        ID,
        NAME,
        CITY,
        SALESREP_ID,
        CURRENCY_ID,
        ACTIVE_FLAG,
        LAST_ORDER_DATE
    FROM CUSTOMER
    WHERE NAME LIKE '%' + @SearchTerm + '%' 
       OR ID LIKE '%' + @SearchTerm + '%'
    ORDER BY NAME;
END

CREATE PROCEDURE sp_GetOutstandingPayments
    @DateFrom DATE = NULL,
    @DateTo DATE = NULL,
    @Offset INT = 0,
    @Limit INT = 50
AS
BEGIN
    SET NOCOUNT ON;
    
    DECLARE @SQL NVARCHAR(MAX) = '
    SELECT
        t.ID,
        t.CUSTOMER_PO_REF,
        t.CONTACT_FIRST_NAME,
        t.SHIP_VIA,
        t.SALESREP_ID,
        t.FREIGHT_TERMS,
        t.TERMS_DESCRIPTION,
        t.STATUS,
        t.TOTAL_AMT_ORDERED,
        t.TOTAL_AMT_SHIPPED,
        t.AMT_OUTSTANDING,
        co.ORDER_DATE,
        co.CURRENCY_ID,
        c.NAME AS CUSTOMER_NAME
    FROM (
        SELECT 
            ID,
            CUSTOMER_PO_REF,
            CONTACT_FIRST_NAME,
            SHIP_VIA,
            SALESREP_ID,
            FREIGHT_TERMS,
            TERMS_DESCRIPTION,
            STATUS,
            TOTAL_AMT_ORDERED,
            TOTAL_AMT_SHIPPED,
            (ISNULL(TOTAL_AMT_ORDERED, 0) - ISNULL(TOTAL_AMT_SHIPPED, 0)) AS AMT_OUTSTANDING
        FROM CUSTOMER_ORDER
    ) t
    LEFT JOIN CUSTOMER_ORDER co ON t.ID = co.ID
    LEFT JOIN CUSTOMER c ON co.CUSTOMER_ID = c.ID
    WHERE t.AMT_OUTSTANDING > 0';
    
    DECLARE @Params NVARCHAR(MAX) = '';
    
    IF @DateFrom IS NOT NULL AND @DateTo IS NOT NULL
    BEGIN
        SET @SQL = @SQL + ' AND co.ORDER_DATE BETWEEN @DateFromParam AND @DateToParam';
        SET @Params = '@DateFromParam DATE, @DateToParam DATE';
    END
    ELSE IF @DateFrom IS NOT NULL
    BEGIN
        SET @SQL = @SQL + ' AND co.ORDER_DATE >= @DateFromParam';
        SET @Params = '@DateFromParam DATE';
    END
    ELSE IF @DateTo IS NOT NULL
    BEGIN
        SET @SQL = @SQL + ' AND co.ORDER_DATE <= @DateToParam';
        SET @Params = '@DateToParam DATE';
    END
    
    SET @SQL = @SQL + ' ORDER BY co.ORDER_DATE DESC OFFSET @OffsetParam ROWS FETCH NEXT @LimitParam ROWS ONLY';
    
    IF @Params = ''
        SET @Params = '@OffsetParam INT, @LimitParam INT';
    ELSE
        SET @Params = @Params + ', @OffsetParam INT, @LimitParam INT';
    
    EXEC sp_executesql 
        @SQL, 
        @Params,
        @DateFromParam = @DateFrom,
        @DateToParam = @DateTo,
        @OffsetParam = @Offset,
        @LimitParam = @Limit;
END

CREATE PROCEDURE sp_SearchGLAccounts
    @SearchTerm NVARCHAR(100),
    @StartYear INT,
    @StartPeriod INT,
    @EndYear INT,
    @EndPeriod INT
AS
BEGIN
    SET NOCOUNT ON;
    
    SELECT
        a.ID                        AS Account_ID,
        a.DESCRIPTION               AS Description,
        ab.ACCT_YEAR                AS Fiscal_Year,
        ab.ACCT_PERIOD              AS Fiscal_Period,
        FORMAT(DATEFROMPARTS(
            CASE WHEN ab.ACCT_PERIOD >= 11 THEN ab.ACCT_YEAR ELSE ab.ACCT_YEAR - 1 END,
            CASE ab.ACCT_PERIOD
                WHEN 1 THEN 3 WHEN 2 THEN 4 WHEN 3 THEN 5 WHEN 4 THEN 6
                WHEN 5 THEN 7 WHEN 6 THEN 8 WHEN 7 THEN 9 WHEN 8 THEN 10
                WHEN 9 THEN 11 WHEN 10 THEN 12 WHEN 11 THEN 1 WHEN 12 THEN 2
            END, 1), 'MMM/yyyy')    AS Period,
        ab.CURR_BALANCE             AS Balance,
        (ab.DEBIT_BUDGET - ab.CREDIT_BUDGET) AS Budget,
        SUM(ab.DEBIT_BUDGET - ab.CREDIT_BUDGET)
            OVER (PARTITION BY ab.ACCOUNT_ID, ab.ACCT_YEAR ORDER BY ab.ACCT_PERIOD)
                                    AS Cumulative_Budget,
        ab.CURR_BALANCE - (ab.DEBIT_BUDGET - ab.CREDIT_BUDGET) AS Variance
    FROM ACCOUNT a
    INNER JOIN ACCOUNT_BALANCE ab ON a.ID = ab.ACCOUNT_ID
    WHERE (a.ID LIKE '%' + @SearchTerm + '%' OR a.DESCRIPTION LIKE '%' + @SearchTerm + '%')
      AND (
            (ab.ACCT_YEAR > @StartYearParam OR (ab.ACCT_YEAR = @StartYearParam AND ab.ACCT_PERIOD >= @StartPeriodParam))
        AND (ab.ACCT_YEAR < @EndYearParam OR (ab.ACCT_YEAR = @EndYearParam AND ab.ACCT_PERIOD <= @EndPeriodParam))
      )
    ORDER BY a.ID, ab.ACCT_YEAR, ab.ACCT_PERIOD;
END
GO

CREATE PROCEDURE sp_CheckOrderCompletion
    @OrderID NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;
    
    SELECT 
        @OrderID AS OrderID,
        CASE WHEN EXISTS (
            SELECT 1 FROM CUST_ORDER_LINE col 
            WHERE (col.CUST_ORDER_ID = @OrderID OR col.CUST_ORDER_ID = '`' + @OrderID)
            AND col.ORDER_QTY > ISNULL(col.TOTAL_SHIPPED_QTY, 0)
        ) THEN 0 ELSE 1 END AS IsComplete,
        (SELECT COUNT(*) FROM CUST_ORDER_LINE 
         WHERE (CUST_ORDER_ID = @OrderID OR CUST_ORDER_ID = '`' + @OrderID)) AS TotalLines,
        (SELECT COUNT(*) FROM CUST_ORDER_LINE 
         WHERE (CUST_ORDER_ID = @OrderID OR CUST_ORDER_ID = '`' + @OrderID)
         AND ORDER_QTY <= ISNULL(TOTAL_SHIPPED_QTY, 0)) AS CompletedLines;
END
GO

CREATE PROCEDURE sp_GetCustomerOrderSummary
    @CustomerID NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;
    
    SELECT 
        COUNT(*) AS TotalOrders,
        SUM(TOTAL_AMT_ORDERED) AS TotalOrderedAmount,
        SUM(TOTAL_AMT_SHIPPED) AS TotalShippedAmount,
        SUM(ISNULL(TOTAL_AMT_ORDERED, 0) - ISNULL(TOTAL_AMT_SHIPPED, 0)) AS OutstandingAmount,
        AVG(TOTAL_AMT_ORDERED) AS AverageOrderValue,
        MAX(ORDER_DATE) AS LastOrderDate,
        MIN(ORDER_DATE) AS FirstOrderDate
    FROM CUSTOMER_ORDER
    WHERE CUSTOMER_ID = @CustomerID;
END
GO
